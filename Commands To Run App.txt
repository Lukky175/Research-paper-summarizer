// To Build and Run This Application You must be on Linux Environment With Docker Installed

//First Install Docker Then Download ZIP of this repo or Pull the repo

//To run The Application you Just Need One Command
docker compose up --build

// To stop the application
docker-compose down
or
docker-compose down -v (To also Delete Docker Volume)







// To manuall Run This Project Without Docker

//For Backend
cd backend
npm init -y  //(To install package.json)
cd backend
npm install
sudo systemctl start mongod   //start mongodb
node server.js


// For Frontend
cd frontend
npm init -y
npm install


cd webapp3tier
cd backend
npm start


//To check Entries of MongoDB

cd webapp3tier

sudo systemctl start mongod

mongosh
